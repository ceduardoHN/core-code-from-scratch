```
function firstWord(string){
   let firstSpace=string.indexOf(" ");
   let word=string.substr(0,firstSpace);
   return word;
}
```

```
function normalize(dateString){
   let newDate=dateString.replace("-","/").replace("-","/");
   return newDate;
}
```

```
}
```
